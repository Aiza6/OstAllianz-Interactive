// Initialize the map
var map = L.map('map').setView([48.1351, 11.5820], 11); // Centered on Munich region

// Add OpenStreetMap as the base layer
L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
    attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors',
    maxZoom: 19
}).addTo(map);

// Load and display the Parking GeoJSON file
fetch('data/Parking 1.geojson') // Path to your GeoJSON file
    .then(response => response.json()) // Read the file content
    .then(data => {
        // Add the GeoJSON layer to the map
        L.geoJSON(data, {
            style: { color: 'blue', weight: 2 }, // Style for the parking areas
            onEachFeature: function (feature, layer) {
                // Bind a popup to each feature
                if (feature.properties) {
                    layer.bindPopup("Parking: " + feature.properties.name || "Unnamed");
                }
            }
        }).addTo(map);
    })
    .catch(error => console.error("Error loading GeoJSON:", error));
